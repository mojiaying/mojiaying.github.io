                              # 组件传值

                              # 1. 父传子：props

                              1. 父组件提供要传递的数据
                              2. 给子组件标签添加属性，属性值为要传递的数据
                              3. 子组件通过props接收父组件中传递的数据
                              // 1. 类组件式
export default class Main extends Component {
  constructor() {
    super(props)
  }
  render() {
    const { list, name } = this.props
    return (
      <div>
        <h1>{name}</h1>
        <ul>
          {list.map(item => (
            <li key={item}>{item}</li>
          ))}
        </ul>
      </div>
    )
  }
}
// 2. 函数组件式
export default function Main(props) {
  return (
    <div>
      <h1>main</h1>
      <h2>{props.name}</h2>
    </div>
  )
}

                              # 2. 属性类型校验

                              对于父组件传给子组件的数据，有时候是需要我们校验

                              可以用以下方法校验属性类型

                              1. Flow（代码静态类型的检查工具）
                              2. TS
                              3. prop-types 库 来进行属性校验
                              yarn add -D prop-types # 属性校验的库
# react 15.5之前 React.PropTypes prop-types 

                              import PropTypes from 'prop-types'
// 子组件 需要接收属性 还要做属性校验 propTypes
class Son extends Component {
  constructor(props) {
    super(props)
  }

  render() {
    return (
      <div>
        <h3>我是子组件</h3>
        <h3>我有{this.props.money}元</h3>
        <ul>
          {this.props.wealth.map(item => (
            <li key={item}>{item}</li>
          ))}
        </ul>
        <span>
          {/* 传递过来的女朋友 */}
          {this.props.name}
          {this.props.age}
          {this.props.brand}
          {this.props.color}
        </span>
        <h1>{JSON.stringify(this.props.info)}</h1>
      </div>
    )
  }
}

// 子组件添加一个propTypes属性
Son.propTypes = {
  // money 传递的必须是number 必须传递的属性
  money: PropTypes.number.isRequired,
  // 传递的类型必须是数组
  wealth: PropTypes.array,
  car: PropTypes.object,
  // 对象的某个属性的值进行校验的话
  age: PropTypes.number
}

// 默认值
Son.defaultProps = {
  money: 100000,
  wealth: ['默认值知识', '默认值品德']
}

                              # 3. 子传父

                              思路：利用了回调函数，父组件提供回调，子组件调用，将要传递的数据作为回调函数的参数

                              1. 父组件提供一个函数（用于接收数据）
                              2. 将这个函数作为属性的值，传递给子组件
                              3. 子组件通过props调用父组件的函数
                              4. 将子组件中的数据作为参数传递给父组件的函数
                              // 父组件
 render() {
    const { number } = this.state
    return (
      <div>
        <h1>父组件当前的number:{number}</h1>
        <Profile
          customize={n => {
            this.customEvent(n)
          }}
        />
      </div>
    )
  }

  // 实现customEvent
  customEvent(n) {
    this.setState({
      number: this.state.number + n
    })
  }
// 子组件----------------------------------------
 render() {
    return (
      <div>
        <h2>子组件</h2>
        <button onClick={()=>{this.postNumber(this.state.number)}}>点击</button>
      </div>
    )
  }

  postNumber (n) {
    // 接收props里边的customize
    const { customize } = this.props
    // 调用父组件的函数
    customize(n)
  }

                              # 4.跨级组件

                              爷爷组件=>父组件=>孙子组件

                              1. 单向数据流 完整概念是：数据的流向只能通过props由外层到内层一层一层的传递

                              单向数据流和双向数据绑定的区别：双向数据绑定单纯的就是数据与视图层之间的绑定

                              1. 双向数据绑定实现：onChange 回调中更新绑定值
                              render() {
    return (
      <div>
        <h1>{this.state.name}</h1>
        {/* onchang  合成事件 */}
        <input type="text" value={this.state.name} onChange={this.changeVal} />
      </div>
    )
  }
  changeVal = e => {
    this.setState({
      name: e.target.value
    })
  }

                              当我们的组件层次在3层以内我们可以采用props层层传递

                              3.context对象实现跨级传值

                              context就相当于一个全局变量，不管嵌套有多深，可以随意的使用

                              • <context.Provider value={}>提供数据
                              • 使用数据：static contextType = context 通过 this.context访问
                              • 使用数据：<context.Consumer>{value => {return(<></>)}} </context.Consumer>
                              // context.js
const obj = {
  house: '300㎡',
  car: 'bwm'
}

// 为所有的组件创建一个context() 对象，并传入一个默认值 默认值可传可不传
const context = React.createContext(obj)
export default context

// app.js
// 提供数据的组件
render() {
    return (
      <div>
        <context.Provider value={this.state.wealth}>
          <Father />
        </context.Provider>
        <context.Provider value={this.state.wealth}>
          <Son />
        </context.Provider>
        {/* 不传则使用默认值  */}
        <Father/>
      </div>
    )
  }

// 接收数据组件：context.Consumer
function Son() {
  return (
    <context.Consumer>
      {value => {
        return (
          <div>
            <h1>儿子组件</h1>
            <h1>{JSON.stringify(value)}</h1>
            <h2>{value.money}</h2>
          </div>
        )
      }}
    </context.Consumer>
  )
}
// 接收数据组件
class Grandson extends Component {
  // 指定一个context,使用contextType可以通过this.context访问
  static contextType = context

  render() {
    console.log(this.context)
    return (
      <div>
        <h1>孙子组件</h1>
        <h1>{JSON.stringify(this.context)}</h1>
      </div>
    )
  }
}

                              # 5. 兄弟组件使用eventBus

                              通过发送事件把数据传过去

                              yarn add -D events

                              创建对象

                              import { EventEmitter } from 'events'

// 事件总线 可以利用这个对象发射和监听事件，这个对象是全局的
const event = new EventEmitter()

export default event

                              发送事件

                              import event from '../event'
 render() {
    return (
      <div>
        <h1>哥哥组件</h1>
        <button onClick={this.emitter}>发送事件</button>
      </div>
    )
  }

  emitter = () => {
    // 发送一个事件
    event.emit('giveMoney', this.state)
  }

                              接收事件

                              import event from '../event'
  // 添加事件监听
  componentDidMount() {
    event.addListener('giveMoney', this.giveMoneyHandler)
  }
  // 处理事件监听函数
  giveMoneyHandler = obj => {
    this.setState({
      money: obj.money,
      glod: obj.glod
    })
  }

  // 移除事件监听
  componentWillUnmount() {
    event.removeAllListeners('giveMoney',this.giveMoneyHandler)
  }
  render() {
    return (
      <div>
        <h1>弟弟组件</h1>
        <hr />
        <h2>{this.state.money}</h2>
        <h2>{this.state.glod}</h2>
      </div>
    )
  }
