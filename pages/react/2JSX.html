                        # JSX 语法

                        # ES6 的类

                        class Person {
  // 构造函数
  constructor (name,money) {
    this.name = name
    this.money = money
  }

  // 类方法
  say() {
    console.log(`我是${this.name},我有${this.money}元`)
  }
}
// 实例化对象
const zhangsan = new Person('张三',10000)
zhangsan.say()

// 类继承
class Lisi extends Person {
  constructor(name,money,age) {
    // 作为函数调用，调用的是父类的构造函数
    super(name,money)
    this.age = age
  }

  study() {
    console.log(`${this.name}说：好好学习天天向上`)
  }

  speak() {
    console.log(`我今年${this.age}`)
  }
}


const lisi = new Lisi('李四',20000,20)
lisi.say()
lisi.study()
lisi.speak()

                        # 1.jsx的作用

                        react的作用：监听数据更新，帮助我们去渲染视图

                        render()函数实现了DOM的渲染的过程，这个过程我们可以分为两个阶段：

                        1. 获取虚拟DOM（JSX经过babel转译后的对象就是虚拟DOM）
                        2. 将虚拟DOM转化成真实DOM，并插入到HTML文档中

                        JSX javascript xml (xml 一般指的就是可扩展标记语言)

                        jsx是react提供的语法糖（syntax sugar）,能让我们在JS中去写HTML，在React中它的作用是描述UI，可以把它理解为虚拟DOM

                        # 2.jsx的基本语法

                        1. JSX只能有一个根标签
                        2. 不需要使用引号包裹
                        3. {/**/} 注释
                        4. {this.state.name} 插值表达式
                        5. jsx中可以是单标签，也可以是双标签，单标签必须闭合
                        6. null undefined boolean 不能被渲染的，如果需要显示，需要转换字符串test + ''或者 test.toString()
                        <h1>{this.state.test+''}</h1>
<h1>{this.state.message.toString()}</h1>
                        1. 对象使用JSON.stringify(obj)或许obj.属性名
                        <h1>{this.state.info.name}</h1>
<h2>{JSON.stringify(this.state.info)}</h2>
                        1. 迭代数据渲染用for或者 map
                        <ul>
  {
  this.state.hobby.map((item, index) => {
    return (
      <li key={item}>
        <span>{index}</span>
        <span>{item}</span>
      </li>
    )
  })
  }
</ul>
                        1. jsx中可以插入表达式，但是不可以插入语句：JSX中不能用if语句，可以用三元表达式，表达式最终会有返回值
                        <div>
  {/*三元表达式*/}
  <h1>{this.state.status ? '已登录' : '未登录'}</h1>
  {/*jsx中不能写if语句*/}
</div>
                        1. 数据渲染：在render把数据结构出来 ,可以通过函数调用把数据渲染出来
                         render() {
    // 解构
    const {name,age} = this.state
    return (
      <div>
        <h1>{name}</h1>
        <h1>{age}</h1>
        {/*函数调用*/}
        <h1>{this.say()}</h1>
      </div>

    )
  }
  say() {
    return `${this.state.name},${this.state.age}`
  }

                        1. 动态绑定属性 使用三元表达式
                        <div className={"div1 " + (active ? 'active' : '')}>
  我是div
</div>
<button onClick={() => { this.change() }}>按钮</button>

change() {
  this.setState({
    active: !this.state.active
  })
}
                        1. 条件渲染： 一、用变量接收内容，插值渲染 二、用三元表达式 三、用逻辑语
                        render() {
        const { status } = this.state
        let flag = null
        let btnTxt = null
        // 1.用变量接收内容，插值渲染
        if (status) {
          flag = <h2>已经登录</h2>
          btnTxt = '退出'
        } else {
          flag = <h2>请登录</h2>
          btnTxt = '登录'
        }
        return (
          <div>
            {flag}
            <button>{btnTxt}</button>
            <button onClick={() => { this.changStatus() }}>
              {/* 2.三元表式 */}
              {status ? '退出' : '登录'}
            </button>
            <hr />
            <h2>{status ? '我不重要' : null}</h2>


            {/*3. 逻辑与,一个条件不成立,后面的条件都不成立*/}
            <h2>{status && '大帅逼2号'}</h2>
            {/*当status为true的时候,整个大帅逼2号会显示,为false的时候就不显示*/}
            <h2>{status && <h2>大帅逼2号</h2>}</h2>

          </div>

        )
      }

      changStatus() {
        this.setState({
          status: !this.state.status
        })
      }

                        v-show 实现方法

                        render() {
  const { status } = this.state
  const displayVal = status ? 'block' : 'none'

                        1. 操作数组：for循环或map
                        render() {
        const { name, number } = this.state
        // 1. for循环是语句不是表达式 不能写在jsx中 if也是语句 表达式有返回值
        const items = []
        for (let i = 0; i < name.length; i++) {
          items.push(<li key={i}>{name[i]}</li>)
        }

        return (
          <div>
            <h2>数组可以直接显示</h2>
            {name}
            <h2>1.for循坏</h2>
            <ul>{items}</ul>
            <h2>2. map或数组方法 （有返回值）</h2>
            <ul>
              {
                name.map((item, index) => {
                  return <li key={index} style={{ listStyle: 'none' }}>{item}</li>
                })
              }
            </ul>
            {/*筛选大于等于50的数字,并遍历*/}
            {
              number.filter(item => item >= 50).map(
                item => <li key={item}>{item}</li>
              )
            }

          </div>

        )
      }
